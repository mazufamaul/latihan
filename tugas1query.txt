1. masuk ke ruang DB melalui CMD 
----------------------------------------
C:\Users\USER>cd \xampp\mysql\bin
C:\xampp\mysql\bin>mysql -u root -p


---------------------------------------
2. tampilkan database yang sudah dibuat
--------------------------------------
MariaDB [(none)]> use stupen;

MariaDB [stupen]>


-----------------------------------------
3. buat database baru : dbpegawai
-----------------------------------------
MariaDB [stupen]> create database dbpegawai;
MariaDB [stupen]> use dbpegawai;


----------------------------------------
4. buat tabel pegawai field-field:
----------------------------------------

MariaDB [dbpegawai]> create table pegawai (
    -> id INT PRIMARY KEY,
    -> nim CHAR(5) NOT NULL UNIQUE,
    -> nama VARCHAR(30) NOT NULL,
    -> gender ENUM('L','P') NOT NULL,
    -> umur INT NOT NULL,
    -> berat_badan FLOAT
    -> );


CREATE TABLE pegawai (
  id INT PRIMARY KEY,
  nip CHAR(5) NOT NULL UNIQUE,
  nama VARCHAR(30) NOT NULL,
  gender ENUM('L','P') NOT NULL,
  umur INT NOT NULL,
  berat_badan FLOAT
);
